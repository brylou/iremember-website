    <button type="submit">Add Memorial</button>
  </form>

  <div class="cemetery" id="cemetery"></div>

  <script>
    const form = document.getElementById('addForm');
    const cemeteryDiv = document.getElementById('cemetery');

    let graves = JSON.parse(localStorage.getItem('graves')) || [];

    function renderGraves() {
      cemeteryDiv.innerHTML = '';
      graves.forEach((grave, index) => {
        const div = document.createElement('div');
        div.className = 'tombstone';
        div.innerHTML = `
          <h3>ðŸª¦ ${grave.name}</h3>
          <p>"${grave.epitaph}"</p>
          <button class="remove-btn" onclick="removeGrave(${index})">Remove</button>
        `;
        cemeteryDiv.appendChild(div);
      });
    }

    form.addEventListener('submit', function(e) {
      e.preventDefault();
      const name = document.getElementById('name').value.trim();
      const epitaph = document.getElementById('epitaph').value.trim();
      if (name && epitaph) {
        graves.push({ name, epitaph });
        localStorage.setItem('graves', JSON.stringify(graves));
        renderGraves();
        form.reset();
      }
    });

    function removeGrave(index) {
      graves.splice(index, 1);
      localStorage.setItem('graves', JSON.stringify(graves));
      renderGraves();
    }

    // Initial render
    renderGraves();
  </script>
